create table FacilityItems(
	FacilityItemId int primary key auto_increment,
	Title varchar(63),
	Info varchar(255),
	Positionx double(16,8),
	Positiony double(16,8),
	Price float,
	Time varchar(63)
);

create table CanteenItems(
	CanteenItemId int primary key auto_increment,
	Name varchar(31),
	FacilityItemId int,
	Info varchar(255),
	Price float,
	SpicyRate float,
	SweetRate float,
	Mark float
);

create table Images(
	ImageId int primary key auto_increment,
	ImgLoc varchar(255),
	CanteenItemId int,
	IsMainImg int
);

create table Opinions(
	Id int primary key auto_increment,
	Content varchar(255),
	Solution varchar(255),
	Type int,
	Result int
);

create table Comments(
	Id int primary key auto_increment,
	UserId int ,
	CanteenItemId int,
	Content varchar(255),
	mark float,
	isLike int
);

create table Users(
	Id int primary key auto_increment,
	UserName varchar(15),
	Password varchar(31),
	Level int
);
